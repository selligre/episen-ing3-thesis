\section{Structural Limitations of Kernel-Level Anti-Cheats}
\label{sec:two:one}

\subsection{Technological Arms Race}
\label{sec:two:one:one}

\begin{itemize}
\item Constant adaptation of cheaters
\item Increasing development costs
\item Dependency on continuous updates
\end{itemize}

\subsection{Limits Against New Forms of Cheating}
\label{sec:two:one:two}

\begin{itemize}
\item Cheat externalization outside the machine
\item Hardware-assisted cheating
\end{itemize}

\subsection{Economic and Operational Costs}
\label{sec:two:one:three}

\begin{itemize}
\item Driver maintenance and technical support
\item Security incident management
\item Reputational impact in case of issues
\end{itemize}
